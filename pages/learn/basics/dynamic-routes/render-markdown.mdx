要渲染 markdown 内容，我们将使用 `remark` 库。首先，让我们安装它:

```shell
npm install remark remark-html
```

在 `lib/posts.js`中导入他们:

```js
import remark from 'remark'
import html from 'remark-html'
```

并将 `getPostData()` 更新如下，以使用`remark`。

```js
export async function getPostData(id) {
  const fullPath = path.join(postsDirectory, `${id}.md`)
  const fileContents = fs.readFileSync(fullPath, 'utf8')

  // Use gray-matter to parse the post metadata section
  const matterResult = matter(fileContents)

  // Use remark to convert markdown into HTML string
  const processedContent = await remark()
    .use(html)
    .process(matterResult.content)
  const contentHtml = processedContent.toString()

  // Combine the data with the id and contentHtml
  return {
    id,
    contentHtml,
    ...matterResult.data
  }
}
```

**重要的**: 我们给 `getPostData` 加了 **`async`** 关键字，因为我们需要使用 `await` 来等待 `remark` 执行完成。

这意味着我们需要更新 `pages/posts/[id].js` 中的 `getStaticProps`，以便在调用 `getPostData` 时使用 `await`:

```js
export async function getStaticProps({ params }) {
  // Add the "await" keyword like this:
  const postData = await getPostData(params.id)
  // ...
}
```

最后，更新 `Post` 组件，使用 `dangerouslySetInnerHTML` 渲染 `contentHtml`:

```jsx
export default function Post({ postData }) {
  return (
    <Layout>
      {postData.title}
      <br />
      {postData.id}
      <br />
      {postData.date}
      <br />
      <div dangerouslySetInnerHTML={{ __html: postData.contentHtml }} />
    </Layout>
  )
}
```
